#include <stdio.h>
#include <locale.h>
#include <math.h>

int main()
{
    int a = 0, b = 0, c = 0, i = 21, j = 21, l = 21; 
    do{
        if (i % 7 == 0) 
        a += i;
        i++;
        
    } 
    while (i < 100);
    for(j=21; j < 100; j++)
    {
        if (j % 7 == 0) 
        b += j;
        
    }
    while(l<100)
    {
        if (l % 7 == 0) 
        c += l;
        l++;
    }
    printf("a=%d \n",a);
    printf("b=%d \n",b);
    printf("c=%d \n",c);
    return 0;
}
